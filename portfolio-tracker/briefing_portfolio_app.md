  id           SERIAL PRIMARY KEY,\
  code         VARCHAR(20) NOT NULL UNIQUE,  -- ex: "PETR4", "BTC", "CDB C6"\
  name         VARCHAR(100) NOT NULL,\
  class        VARCHAR(30) NOT NULL,  -- FIIs | A\'e7\'f5es Brasil | Cripto | Renda Fixa | Exterior | Previd\'eancia\
  sector       VARCHAR(50),\
  country      VARCHAR(20),           -- Brasil | USA | Fran\'e7a\
